CREATE TABLE ORDERS
(ORDER_ID NUMBER(10), ORDER_DATE DATE, CUSTOMER_NAME VARCHAR2(20) CONSTRAINT NN_CUNAME NOT NULL, PAYMENT_TYPE VARCHAR2(10),
DELIEVERY_METHOD VARCHAR2(20), ADDRESS VARCHAR2(20), TOTAL_PRICE NUMBER(20),
CONSTRAINT CK_PAYMENT CHECK(PAYMENT_TYPE IN ('CREDIT', 'CASH')),
CONSTRAINT PK_ORDER_ID PRIMARY KEY (ORDER_ID),
CONSTRAINT FK_ORDER_ID  FOREIGN KEY(ORDER_ID) REFERENCES ORDERS(ORDER_ID));


CREATE TABLE ORDER_ITEMS
(ITEM_ID NUMBER(10), ITEM_NAME VARCHAR2(20), ITEM_DESC VARCHAR2(10), ITEM_SIZE NUMBER(10), ITEM_PRICE NUMBER(20),
EXPIRE_DATE DATE CONSTRAINT NN_EXPIRE_DATE NOT NULL, INVENTORY NUMBER(10), ORDER_ID NUMBER(10),
CONSTRAINT CK_ITEM_PRICE CHECK(ITEM_PRICE BETWEEN 100 AND 300),
CONSTRAINT PK_ITEM_ID PRIMARY KEY (ITEM_ID));
